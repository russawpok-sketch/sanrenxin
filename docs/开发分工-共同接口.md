## 类型定义（`shared/types.ts`）

```typescript
export type AnimalType = 'rooster' | 'monkey' | 'dog' | 'duck';

export interface TransformEvent {
  animal: AnimalType;
  confidence: number;   // 0~1，仅供 UI 展示
  roastText: string;    // 「你像加班后的狗」
  timestamp: number;
}

export const ANIMAL_LABELS: Record<AnimalType, string> = {
  rooster: '公鸡',
  monkey: '猴子',
  dog: '狗',
  duck: '鸭子',
};

export const ANIMAL_HINTS: Record<AnimalType, string> = {
  rooster: '喔喔喔',
  monkey: '啊啊啊',
  dog: '汪汪',
  duck: '嘎嘎嘎',
};
```

## 事件总线（`shared/events.ts`）

```typescript
import type { TransformEvent } from './types';

type TransformListener = (event: TransformEvent) => void;

const listeners: TransformListener[] = [];

/** A 线调用：识别成功后发布 */
export function emitTransform(event: TransformEvent): void {
  listeners.forEach((fn) => fn(event));
}

/** B 线调用：订阅变身事件 */
export function onTransform(callback: TransformListener): void {
  listeners.push(callback);
}
```

## Mock 联调方法

**A 线**（Day 1~2 开发期间）：

```typescript
// 键盘 fallback，不依赖麦克风
document.addEventListener('keydown', (e) => {
  const map: Record<string, AnimalType> = {
    '1': 'rooster', '2': 'monkey', '3': 'dog', '4': 'duck',
  };
  const animal = map[e.key];
  if (animal) {
    emitTransform({
      animal,
      confidence: 0.9,
      roastText: `[mock] 识别为${ANIMAL_LABELS[animal]}`,
      timestamp: Date.now(),
    });
  }
});
```

**B 线**（Day 1~2 开发期间）：

```typescript
onTransform((event) => {
  player.setAnimal(event.animal);
  gameUI.showRoast(event.roastText);
});
```
